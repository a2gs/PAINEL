Fazer uma funcao como:

#include <stdlib.h>
int loadPAINEL_HOMEwith(char *fullpathOut, size_t fullpathOutSz, char *path)
{
	char *c = NULL;
	c = getenv(PAINEL_HOME);
	if(c == NULL){
		return(NOK);
	}
	snprintf(fullpathOut, fullpathOutSz, "%s/%s", c, path);

	return(OK);
}
